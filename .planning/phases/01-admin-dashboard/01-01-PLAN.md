Create a price calculator utility that implements the pricing formula per D-08 (CONTEXT.md):
- Base formula: `price = width × height × base_price + extras`
- Extras can be: fixed amount (flat fee) OR per-m² (scaled by area)
- Used client-side (no API calls during pricing)
- Exported for use in product forms and Phase 2 lead form

Implementation:
```typescript
// src/lib/utils/price-calculator.ts
export interface ProductExtra {
  type: 'fixed' | 'per-m2';
  amount: number;
}

/**
 * Calculate product price based on dimensions and base price + extras.
 * Formula: (width × height × base_price) + extras
 * 
 * @param width - Width in meters (decimal)
 * @param height - Height in meters (decimal)
 * @param basePrice - Base price per m²
 * @param extras - Optional array of extra charges (fixed or per-m²)
 * @returns Calculated price (rounded to nearest whole number)
 */
export function calculatePrice(
  width: number,
  height: number,
  basePrice: number,
  extras: ProductExtra[] = [],
): number {
  // Handle invalid inputs
  if (width <= 0 || height <= 0 || basePrice <= 0) {
    return 0;
  }

  // Calculate area in m²
  const area = width * height;

  // Start with base price calculation
  let totalPrice = area * basePrice;

  // Add extras
  for (const extra of extras) {
    if (extra.type === 'fixed') {
      totalPrice += extra.amount;
    } else if (extra.type === 'per-m2') {
      totalPrice += area * extra.amount;
    }
  }

  // Round to nearest whole number (currency standard)
  return Math.round(totalPrice);
}

/**
 * Format price as Vietnamese currency (VND, no decimal places)
 */
export function formatPrice(price: number): string {
  return new Intl.NumberFormat('vi-VN', {
    style: 'currency',
    currency: 'VND',
    maximumFractionDigits: 0,
  }).format(price);
}
```

Ensure:
- Function is pure (no side effects)
- Handles edge cases (0 or negative dimensions)
- Returns number (not string)
- Includes JSDoc comments
- Can be used in client components (no dependencies on server APIs)
